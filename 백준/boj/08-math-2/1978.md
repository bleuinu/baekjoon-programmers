# 1978. 소수 찾기


[problem](https://www.acmicpc.net/problem/1978)

```cpp
#include <bits/stdc++.h>
using namespace std;

bool isPrime(int n) {
	if(n<2) return false;
	if(n<=3) return true;
	if(n%2==0 || n%3==0) return false;

	for(int i=5; i*i<=n; i+=6) {
		if(n%i==0 || n%(i+2)==0) return false;
	}
	return true;
}

int main() {
	int n;
	cin >> n;
	int cnt = 0;
	for(int i=0; i<n; ++i) {
		int t;
		cin >> t;
		if(isPrime(t)) ++cnt;
	}
	cout << cnt << endl;

	return 0;
}
```
