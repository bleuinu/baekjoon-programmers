# 2231. 분해합

[problem](https://www.acmicpc.net/problem/2231)

```cpp
#include <iostream>
#include <vector>
using namespace std;

int d(int x) {
	int sum = x;
	while(x>0) {
		sum += x%10;
		x/=10;
	}

	return sum;
}

int main() {
	int n;
	cin >> n;

	int i = 1;
	while(true) {
		if(i > n) {
			cout << 0 << endl;
			break;
		}
		if(d(i) == n) {
			cout << i << endl;
			break;
		}
		++i;
	}

	return 0;
}
```
