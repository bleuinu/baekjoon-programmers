# 2798. blacjack

[problem](https://www.acmicpc.net/problem/2798)

```cpp
#include <iostream>
using namespace std;

int main() {
	int n, m, sum=0;
	cin >> n >> m;

	int arr[n];
	for(int i=0; i<n; ++i) {
		cin >> arr[i];
	}

	for(int i=0; i<n; ++i) {
		for(int j=i+1; j<n; ++j) {
			for(int k=j+1; k<n; ++k) {
				int t = arr[i] + arr[j] + arr[k];
				if(t <= m && t > sum) {
					sum = t;
				}
			}
		}
	}

	cout << sum << endl;

	return 0;
}
```
