# 4344. 평균은 넘겠지

[problem](https://www.acmicpc.net/problem/4344)

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	scanf("%d", &t);
	for(int j=0; j<t; ++j) {
		int n;
		int sum = 0;
		double avg = 0.0;
		scanf("%d", &n);
		int arr[n];
		for(int i=0; i<n; ++i) {
			scanf("%d", &arr[i]);
			sum += arr[i];
		}
		avg = sum / (double)n;

		int cnt = 0;
		for(int i=0; i<n; ++i) {
			if(arr[i] > avg) ++cnt;
		}
		avg = cnt / (double)n;
		printf("%.3f%%\n", avg*100);
	}

	return 0;
}
```
