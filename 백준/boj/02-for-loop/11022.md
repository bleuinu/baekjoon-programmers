#. 11022 A+B - 8

[problem](https://www.acmicpc.net/problem/11022)

```cpp
#include <bits/stdc++.h> 
using namespace std;
int main(void) {
  int t, a, b;
  scanf("%d", &t);

  for(int i=1; i<=t; ++i) {
    scanf("%d %d", &a, &b);
    printf("Case #%d: %d + %d = %d\n", i, a, b, a+b);
  }

  return 0;
}
```
