# 2751. 수 정렬하기 2

[problem](https://www.acmicpc.net/problem/2751)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define ios cin.tie(NULL), ios_base::sync_with_stdio(false)
#define endl '\n'

int main() {
	ios;
	int n;
	cin >> n;
	vector<int> v(n,0);
	for(int i=0; i<n; ++i) {
		cin >> v[i];
	}

	sort(v.begin(), v.end());

	for(int i=0; i<n; ++i)
		cout << v[i] << endl;

	return 0;
}
```
