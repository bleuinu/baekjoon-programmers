# 1065. 한수

[problem](https://www.acmicpc.net/problem/1065)

```cpp
#include <iostream>
using namespace std;

int hansoo(int n) {
	if(n < 100) return n;
	if(n <= 110) return 99;

	int cnt = 99;
	for(int i=111; i<=n; ++i) {
		int t = i;
		int c = t%10;
		t/=10;
		int b = t%10;
		int a = t/10;

		if(a-b == b-c)  ++cnt;
	}

	return cnt;
}

int main() {
	int n; 
	cin >> n;
	cout << hansoo(n) << endl;
	return 0;
}
```
