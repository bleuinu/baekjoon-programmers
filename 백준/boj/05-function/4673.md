# 4673. Self Number

[problem](https://www.acmicpc.net/problem/4673)

```cpp
#include <bits/stdc++.h>
using namespace std;

int d(int x) {
	int sum = x;
	while(x>0) {
		sum += x%10;
		x/=10;
	}

	return sum;
}

int main() {
	vector<bool> arr(10001, true);

	for(int i=1; i<10001; ++i) {
		if(!arr[i]) continue;

		int temp = i;
		while(true) {
			int t = d(temp);
			if(t > 10000) break;
			arr[t] = false;
			temp = t;
		}
	}

	for(int i=1; i<10001; ++i) {
		if(arr[i]) cout << i << endl;
	}
	return 0;
}
```
